<div class='mermaid'>
flowchart TD

    %% Super Key Level %%
    A[Super Key\n(Any attribute(s) uniquely identifying rows)] --> B[Candidate Key\n(Minimal super key(s))]

    %% Candidate Key splits %%
    B --> C[Primary Key\n(Chosen candidate key)]
    B --> D[Alternate Key / Unique Key\n(Other candidate keys)]
    B --> E[Composite Key\n(Two or more attributes combined)]

    %% Real Tables %%
    C --> F[STUDENTS Table: student_id PK]
    D --> G[STUDENTS Table: email UK]
    E --> H[ENROLLMENTS Table: (student_id, course_id) PK]

    %% Foreign Key Connections %%
    F --> I[ENROLLMENTS Table: student_id FK]
    J[COURSES Table: course_id PK] --> I
    J --> H

    %% Labels for clarity %%
    classDef pk fill:#9ae6b4,stroke:#2f855a,color:#000
    classDef fk fill:#fbd38d,stroke:#d69e2e,color:#000
    class F,H,J pk
    class I fk
</div>
